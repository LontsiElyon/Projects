81 /**
82 * Registers a player and creates a session for them.
83 * @param username the username of the player.
84 * @param controllerId the controller ID associated with the player.
85 * @param resultHandler handles the result of the registration and session creation.
86 */
87 public void registerPlayerAndCreateSession(String username, String controllerId, Handler<AsyncResult<Void>> resultHandler) {
88 // Check if the controller is already in use
89 logger.info("Check if the controller is already in use: {}",controllerId );
90 checkControllerUsage(controllerId, usageCheckResult -> {
91 if (usageCheckResult.succeeded() && usageCheckResult.result()) {
92 // Controller is already in use, handle the error
93 resultHandler.handle(Future.failedFuture("Controller is already in use by another player"));
94 logger.info("Controller is already in use by another player: {}", controllerId);
95 } else {
96
97 // Call the insertPlayer method to add the player to the database
98 objectRepository.insertPlayer(username, insertPlayerResult -> {
99 if (insertPlayerResult.succeeded()) {
100 // Retrieve the playerId from the result of the insertPlayer operation
101 int playerId = insertPlayerResult.result();
102
103 // Call the createSession method to create a session for the player
104 objectRepository.createSession(playerId, controllerId, "Frontend", createSessionResult -> {
105 if (createSessionResult.succeeded()) {
106 // Call the recordFrontendAssignment method to record the player's assignment to the frontend
107 objectRepository.recordFrontendAssignment(playerId, controllerId, resultHandler);
108 } else {
109 // Handle the failure in creating a session by passing the cause to the result handler
110 resultHandler.handle(Future.failedFuture(createSessionResult.cause()));
111 }
112 });
113 } else {
114 // Handle the failure in inserting a player by passing the cause to the result handler
115 resultHandler.handle(Future.failedFuture(insertPlayerResult.cause()));
116 }
117 });
118 }
119 });
120 }
121 /**
122 * Registers a player using an RFID tag and creates a session.
123 * @param payload the payload containing player information including RFID tag.
124 * @param resultHandler handles the result of the operation.
125 */
126 public void registerPlayerWithRfid(String payload, Handler<AsyncResult<Void>> resultHandler) {
127 JsonObject json = new JsonObject(payload);
128 String rfidTag = json.getString("rfidTag");
129 String controllerId = json.getString("controllerId");
130 String username = json.getString("username");
131
132 // Check if the controller is already in use
133 logger.info("Check if the controller is already in use: {}",controllerId );
134 checkControllerUsage(controllerId, usageCheckResult -> {
135 if (usageCheckResult.succeeded() && usageCheckResult.result()) {
136 // Controller is already in use, handle the error
137 resultHandler.handle(Future.failedFuture("Controller is already in use by another player"));
138 logger.info("Controller is already in use by another player: {}", controllerId);
139 } else {
140 // Proceed with the existing logic to register the player with RFID
141 objectRepository.findPlayerByRfid(rfidTag, findResult -> {
142 if (findResult.succeeded()) {
143 int playerId = findResult.result();
144 if (playerId == -1) {
145 // Player not found, create a new player with the RFID tag
146 objectRepository.insertPlayerWithRfid(username, rfidTag, insertResult -> {
147 if (insertResult.succeeded()) {
148 int newPlayerId = insertResult.result();
149 createRfidSession(newPlayerId, controllerId, rfidTag, resultHandler);
150 } else {
151 resultHandler.handle(Future.failedFuture(insertResult.cause()));
152 }
153 });
154 } else {
155 // Player found, create a session for the existing player
156 createRfidSession(playerId, controllerId, rfidTag, resultHandler);
157 }
158 } else {
159 resultHandler.handle(Future.failedFuture(findResult.cause()));
160 }
161 });
162 }
163 });
164 }

167 /**
168 * Checks if a controller is in use.
169 * @param controllerId the ID of the controller to check.
170 * @param resultHandler handles the result of the usage check.
171 */
172 private void checkControllerUsage(String controllerId, Handler<AsyncResult<Boolean>> resultHandler) {
173 // Check if the controller is in use in the RFIDAssignments table
174 objectRepository.isControllerInUse(controllerId, rfidCheckResult -> {
175 if (rfidCheckResult.succeeded() && rfidCheckResult.result()) {
176 // Controller is in use in RFID assignments
177 resultHandler.handle(Future.succeededFuture(true));
178 } else if (rfidCheckResult.failed()) {
179 // Error in checking RFID assignments
180 resultHandler.handle(Future.failedFuture(rfidCheckResult.cause()));
181 } else {
182 // Controller is not in use in RFID assignments, check FrontendAssignments
183 objectRepository.isControllerInUseInFrontend(controllerId, frontendCheckResult -> {
184 if (frontendCheckResult.succeeded()) {
185 resultHandler.handle(Future.succeededFuture(frontendCheckResult.result()));
186 } else {
187 // Error in checking frontend assignments
188 resultHandler.handle(Future.failedFuture(frontendCheckResult.cause()));
189 }
190 });
191 }
192 });
193 }

216 /**
217 * Generates a random color sequence.
218 * @return the generated color sequence.
219 */
220 public JsonArray generateColorSequence() {
221 // Generate a random sequence length between min and max
222 int sequenceLength = random.nextInt(maxSequenceLength - minSequenceLength + 1) + minSequenceLength;
223
224 JsonArray colorSequence = new JsonArray();
225 for (int i = 0; i < sequenceLength; i++) {
226 String color = COLORS.get(random.nextInt(COLORS.size()));
227 colorSequence.add(color);
228 }
229
230 //Store the generated sequence
231 this.storedSequence = colorSequence;
232 return colorSequence;
233 }

235 /**
236 * Compares the received color sequence with the stored sequence and awards points if they match.
237 * @param controllerId the ID of the controller.
238 * @param receivedSequence the received color sequence.
239 * @return a Future indicating success or failure.
240 */
241 public Future<Boolean> compareSequenceAndAwardPoints(String controllerId, JsonArray receivedSequence) {
242 if (controllerId == null || receivedSequence == null) {
243 logger.error("Invalid input: controllerId or receivedSequence is null");
244 return Future.succeededFuture(false);
245 }
246
247 if (storedSequence == null) {
248 logger.error("No stored sequence available for comparison");
249 return Future.succeededFuture(false);
250 }
251
252 logger.debug("Stored sequence: {}", storedSequence.encode());
253 logger.debug("Received sequence: {}", receivedSequence.encode());
254
255 return objectRepository.fetchPlayerIdByControllerId(controllerId)
256 .compose(playerId -> {
257 if (playerId == null) {
258 logger.error("No player found for controllerId: {}", controllerId);
259 return Future.succeededFuture(false);
260 }
261
262 return objectRepository.fetchCurrentRoundByControllerId(controllerId,playerId)
263 .compose(round -> {
264 if (round == null) {
265 logger.error("Failed to fetch the current round for controllerId: {}", controllerId);
266 return Future.succeededFuture(false);
267 }
268
269 boolean sequencesMatch = compareSequences(storedSequence, receivedSequence);
270
271 if (sequencesMatch) {
272 return objectRepository.updatePoints(controllerId, playerId, round)
273 .map(v -> true)
274 .recover(cause -> {
275 logger.error("Failed to update points for controller {}: {}", controllerId, cause.getMessage());
276 return Future.succeededFuture(false);
277 });
278 } else {
279 return Future.succeededFuture(false);
280 }
281 });
282 });
283 }

285 /**
286 * Compares two sequences of colors.
287 * @param generated the generated color sequence.
288 * @param received the received color sequence.
289 * @return true if the sequences match, false otherwise.
290 */
291 private boolean compareSequences(JsonArray generated, JsonArray received) {
292 if (generated == null || received == null) {
293 logger.error("Either generated or received sequence is null");
294 return false;
295 }
296
297 if (generated.size() != received.size()) {
298 logger.debug("Sequence size mismatch. Generated: {}, Received: {}", generated.size(), received.size());
299 return false;
300 }
301
302 for (int i = 0; i < generated.size(); i++) {
303 String genColor = generated.getString(i);
304 String recColor = received.getString(i);
305 logger.debug("Comparing color at index {}: Generated '{}' vs Received '{}'", i, genColor, recColor);
306 if (!genColor.equalsIgnoreCase(recColor)) {
307 logger.debug("Mismatch at index {}", i);
308 return false;
309 }
310 }
311
312 return true;
313 }
