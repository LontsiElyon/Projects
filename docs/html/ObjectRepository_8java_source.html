570 * @ingroup object_repository

572 *
573 * This method updates the end time of the current round and inserts a new session for the next round,
574 * as well as initializes new display information for that round.
575 *
576 * @param controllerId The unique ID of the controller.
577 * @return A Future indicating the result of the operation.
578 */
579 public Future<Void> createNewRound(String controllerId) {
580 return jdbcPool.withTransaction(client -> {
581 // First, get the current session info
582 String getCurrentSessionQuery = "SELECT player_id, MAX(round) as current_round, login_method FROM Sessions WHERE controller_id = ? AND end_time IS NULL";
583
584 return client.preparedQuery(getCurrentSessionQuery)
585 .execute(Tuple.of(controllerId))
586 .compose(rows -> {
587 if (rows.iterator().hasNext()) {
588 Row row = rows.iterator().next();
589 int playerId = row.getInteger("player_id");
590 int currentRound = row.getInteger("current_round");
591 String loginMethod = row.getString("login_method");
592 int newRound = currentRound + 1;
593
594 // Update the end time of the current round
595 String updateEndTimeQuery = "UPDATE Sessions SET end_time = NOW() WHERE controller_id = ? AND player_id = ? AND round = ?";
596
597 return client.preparedQuery(updateEndTimeQuery)
598 .execute(Tuple.of(controllerId, playerId, currentRound))
599 .compose(v -> {
600 // Insert new session for the new round
601 String insertNewSessionQuery = "INSERT INTO Sessions (controller_id, player_id, round, start_time, login_method) VALUES (?, ?, ?, NOW(), ?)";
602
603 return client.preparedQuery(insertNewSessionQuery)
604 .execute(Tuple.of(controllerId, playerId, newRound, loginMethod));
605 })
606 .compose(v -> {
607 // Insert new DisplayInfo for the new round
608 String insertDisplayInfoQuery = "INSERT INTO DisplayInfo (controller_id, player_id, round, points, username) SELECT ?, ?, ?, 0, user_name FROM Players WHERE player_id = ?";
609
610 return client.preparedQuery(insertDisplayInfoQuery)
611 .execute(Tuple.of(controllerId, playerId, newRound, playerId));
612 });
613 } else {
614 return Future.failedFuture("No active session found for controller: " + controllerId);
615 }
616 })
617 .mapEmpty();
618 });
