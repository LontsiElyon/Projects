23void rfid_check() {
24 // Look for new cards
25 if (!mfrc522.PICC_IsNewCardPresent()) {
26 return;
27 }
28
29 // Select one of the cards
30 if (!mfrc522.PICC_ReadCardSerial()) {
31 return;
32 }
33
34 // Dump UID of the card
35 String rfidTag = "";
36 for (byte i = 0; i < mfrc522.uid.size; i++) {
37 rfidTag += String(mfrc522.uid.uidByte[i] < 0x10 ? "0" : "");
38 rfidTag += String(mfrc522.uid.uidByte[i], HEX);
39 }
40 rfidTag.toUpperCase();
41
42 Serial.println(rfidTag);
43
44 // Print Card type
45 MFRC522::PICC_Type piccType = mfrc522.PICC_GetType(mfrc522.uid.sak);
46 Serial.print("PICC Type: ");
47 Serial.println(mfrc522.PICC_GetTypeName(piccType));
48
49 // Halt PICC
50 mfrc522.PICC_HaltA();
51
52 // Stop encryption on PCD
53 mfrc522.PCD_StopCrypto1();
54
55 // Log and publish the RFID tag
56
57 Serial.print("RFID Tag Detected: ");
58 String payload = "{\"controllerId\":\"" + controllerId + "\", \"rfidTag\":\"" + rfidTag + "\", \"username\":\"Elyon's RFID\"}";
59
60 // Publish the RFID tag
61 if (client.publish("controller/rfid", payload.c_str())) {
62 Serial.println("RFID tag published successfully.");
63 } else {
64 Serial.println("Failed to publish RFID tag.");
65 }
66}
