2/**
3 * @file schema.sql
4 * @brief SQL schema for the RFID-based game system
5 * @details This script creates the necessary tables to store information about controllers, players, sessions, RFID reads, frontend assignments, and display updates.
6 */
7
8/**
9 * @brief Table to store information about controllers
10 * @details This table keeps track of controller statuses and their heartbeat signals.
11 */
12CREATE TABLE Controllers (
13 controller_id VARCHAR(50) PRIMARY KEY, -- Unique identifier for each controller, serving as the primary key
14 status ENUM('online', 'offline') DEFAULT 'offline', -- Status of the controller, with a default value of 'offline'
15 last_heartbeat TIMESTAMP DEFAULT CURRENT_TIMESTAMP, -- Timestamp of the last heartbeat received from the controller
16 created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP -- Timestamp when the controller record was created
17);
18
19/**
20 * @brief Table to store information about players
21 * @details This table stores details about players, including their usernames, RFID tags, and high scores.
22 */
23CREATE TABLE Players (
24 player_id INT AUTO_INCREMENT PRIMARY KEY, -- Unique identifier for each player, auto-incremented
25 user_name VARCHAR(100), -- Name of the player
26 rfid_tag VARCHAR(50) UNIQUE, -- Unique RFID tag assigned to the player
27 high_score INT DEFAULT 0,
28 created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP -- Timestamp when the player record was created
29);
30
31/**
32 * @brief Table to store session details
33 * @details This table logs the player sessions and records which controller is used in each session.
34 */
35CREATE TABLE Sessions (
36 session_id INT AUTO_INCREMENT PRIMARY KEY, -- Unique identifier for each session, auto-incremented
37 player_id INT, -- Foreign key referencing Players table, identifies the player in the session
38 controller_id VARCHAR(50), -- Foreign key referencing Controllers table, identifies the controller used in the session
39 login_method ENUM('RFID', 'Frontend') NOT NULL, -- Method used to login, either via RFID or the frontend
40 start_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP, -- Timestamp when the session started
41 end_time TIMESTAMP NULL, -- Timestamp when the session ended, null if session is ongoing
42 round INT DEFAULT 1, -- The current round within the session, defaulting to 1
43
44 -- Composite unique key
45 UNIQUE KEY unique_session (controller_id, player_id,round),
46
47 -- Foreign key relationships
48 FOREIGN KEY (player_id) REFERENCES Players(player_id), -- Establishes a relationship with the Players table
49 FOREIGN KEY (controller_id) REFERENCES Controllers(controller_id) -- Establishes a relationship with the Controllers table
50);
51
52/**
53 * @brief Table to log RFID read events
54 * @details Stores logs of RFID reads by controllers, including the timestamp and related player information.
55 */
56CREATE TABLE RfidAssignments(
57 log_id INT AUTO_INCREMENT PRIMARY KEY, -- Unique identifier for each RFID log entry, auto-incremented
58 player_id INT, -- Foreign key referencing Players table, identifies the player associated with the RFID log
59 controller_id VARCHAR(50), -- Foreign key referencing Controllers table, identifies the controller where the RFID was read
60 rfid_tag VARCHAR(50), -- The RFID tag read by the controller
61 read_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP, -- Timestamp when the RFID was read
62 FOREIGN KEY (player_id) REFERENCES Players(player_id), -- Establishes a relationship with the Players table
63 FOREIGN KEY (controller_id) REFERENCES Controllers(controller_id) -- Establishes a relationship with the Controllers table
64);
65
66/**
67 * @brief Table to record frontend assignments of players to controllers
68 * @details Logs when a player is assigned to a controller via the frontend interface.
69 */
70CREATE TABLE FrontendAssignments (
71 assignment_id INT AUTO_INCREMENT PRIMARY KEY, -- Unique identifier for each frontend assignment, auto-incremented
72 player_id INT, -- Foreign key referencing Players table, identifies the player in the assignment
73 controller_id VARCHAR(50), -- Foreign key referencing Controllers table, identifies the controller in the assignment
74 assignment_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP, -- Timestamp when the assignment was made
75 FOREIGN KEY (player_id) REFERENCES Players(player_id), -- Establishes a relationship with the Players table
76 FOREIGN KEY (controller_id) REFERENCES Controllers(controller_id) -- Establishes a relationship with the Controllers table
77);
78
79/**
80 * @brief Table to display information
81 * @details This table stores information to be displayed for each controller and player, including points and rounds.
82 */
83CREATE TABLE DisplayInfo (
84 controller_id VARCHAR(50), -- Identifier for the controller
85 player_id INT, -- Foreign key referencing Players table
86 points INT, -- Points to be displayed
87 round INT, -- Round number to be displayed
88 username VARCHAR(50), -- Username associated with the player
89 last_update TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, -- Last update timestamp
90
91 -- Composite primary key to ensure uniqueness
92 PRIMARY KEY (controller_id, round),
93
94 -- Foreign key constraints
95 FOREIGN KEY (controller_id, player_id, round) REFERENCES Sessions(controller_id, player_id, round), -- Reference to Sessions table
96 FOREIGN KEY (player_id) REFERENCES Players(player_id) -- Reference to Players table
97);
98
99
100CREATE TABLE DisplayUpdates (
101 update_id INT AUTO_INCREMENT PRIMARY KEY, -- Unique identifier for each update
102 controller_id VARCHAR(50), -- Foreign key referencing Controllers table
103 update_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP, -- Timestamp of the update
104 FOREIGN KEY (controller_id) REFERENCES Controllers(controller_id) -- Reference to Controllers table
105);
